SELECT GA.IdGame, GA.Title, GA.Description, CO.NameCompany, CO.ActivityCompany, TY.NameType, LI.ContentLink 
FROM game GA
LEFT JOIN have_company HAVECO ON HAVECO.IdGame = GA.IdGame
LEFT JOIN company CO ON CO.IdCompany = HAVECO.idCompany
LEFT JOIN have_type HAVETY ON HAVETY.IdGame = GA.IdGame
LEFT JOIN type TY ON TY.IdType = HAVETY.IdType
LEFT JOIN have_link HAVELI ON HAVELI.IdGame = GA.IdGame
LEFT JOIN link LI ON LI.IdLink = HAVELI.IdLink

SELECT g.TITLE AS 'game',
	g.DESCRIPTION,
	c.NAMECOMPANY,
	t.NAMETYPE,
	l.CONTENTLINK,
	l,IDTYPELINK,
	tl.NAMETYPELINK,

FROM GAME g -- Table de référence
LEFT JOIN HAVE_COMPANY hc ON g.IDGAME = hc.IDGAME -- Jointure entre les tables HAVE_COMPANY et GAME
LEFT JOIN COMPANY c ON hc.IDCOMPANY = c.IDCOMPANY
LEFT JOIN HAVE_TYPE ht ON g.IDGAME = ht.IDGAME -- TYPE
LEFT JOIN TYPE t ON ht.IDTYPE = t.IDTYPE
LEFT JOIN HAVE_LINK hl ON g.IDGAME = hl.IDGAME -- LINK
LEFT JOIN TYPE l ON hl.IDLINK = l.IDLINK
LEFT JOIN HAVE_LINK hl ON g.IDGAME = hl.IDGAME -- LINK
LEFT JOIN TYPE l ON hl.IDLINK = l.IDLINK
LEFT JOIN TYPELINK tl ON l.IDTYPELINK = tl.IDTYPELINK -- TYPELINK
ORDER BY g.IDGAME
